   47     /******************************************************************************
   48      *   Function:  pwm_prescaler
   49        @brief    calculate prescaler .
   50        @details  calculate prescaler
   51   
   52        @param[in]     Freq   -
   53        @param[in]     period      -
   54   
   55        @return  prescaler value
   56      */
   57     static uint16_t pwm_prescaler (uint32_t Freq, uint32_t* period)
   58     {
   59       uint8_t presDiv;
   60       uint8_t mask = 1U;
   61       for (presDiv = 0U; presDiv < 8U; presDiv++) {
   62         *period = (sysClk / mask) / Freq;
   63         if (*period <= 0xFFFF) {
   64           break;
   65         }
   66   
   67         mask = mask << 1U;
   68       }
   69   
   70       return (((uint16_t)presDiv) << 4);
   71     }

  287     /******************************************************************************
  288      *   Function:  pwm_init_564xL_comp
  289        @brief    FlexPWM Init function for compltmentary mode .
  290        @details  FlexPWM Init function for compltmentary mode
  291   
  292        @param[in]     subMod   -
  293        @param[in]     Freq      -
  294        @param[in]     Duty      -
  295        @param[in]     DeadTime      -
  296   
  297        @return  none
  298      */
  299     void pwm_init_564xL_comp(uint8_t pwmMod, uint8_t subMod, uint32_t Freq,
  300       uint16_t resolution, uint16_t DutyA, uint16_t DeadTime, uint16_t outTrig,
  301       uint8_t masterSync, uint8_t pwm45en, uint16_t DutyB, uint16_t OCTRL)
  302     {
  303       volatile mcPWM_tag* FlexPWM = FlexPWM_module[pwmMod];
  304       uint16_t val[6] = { 0U };
  305   
  306       uint16_t init, mask, prescaler;
  307       uint32_t period = 0U;
  308       uint8_t i, iMax;
  309       uint32_t DutyMax = 100 * resolution;
  310       mask = (uint16_t)(0x1U << subMod);
  311       prescaler = pwm_prescaler (Freq, &period);
  312       val[0] = 0;
  313       val[1] = (uint16_t)(period / 2);
  314       init = 0 - val[1];
  315       if (DutyA >= DutyMax) {
  316         val[2] = init;
  317         val[3] = val[1]+1;
  318       } else {
  319         val[3] = (uint16_t)((DutyA * val[1])/DutyMax);
  320         val[2] = 0 - val[3];
  321       }
  322   
  323       if (pwm45en) {
  324         if (DutyB >= DutyMax) {
  325           val[4] = init;
  326           val[5] = val[1]+1;
  327         } else {
  328           val[5] = (uint16_t)((DutyB * val[1])/DutyMax);
  329           val[4] = 0 - val[5];
  330         }
  331   
  332         iMax = 6;
  333       } else {
  334         iMax = 4;
  335       }
  336   
  337       if (masterSync) {
  338         FlexPWM->SUB[subMod].CTRL2.R = 0x9A00;/* Master sync from submodule 0 */
  339       } else {
  340         FlexPWM->SUB[subMod].CTRL2.R = 0x9800;/* Local sync */
  341       }
  342   
  343       FlexPWM->SUB[subMod].INIT.R = init;
  344       for (i = 0; i < iMax; i++) {
  345         FlexPWM->SUB[subMod].VAL[i].R = val[i];
  346       }
  347   
  348       FlexPWM->SUB[subMod].DTCNT0.R = DeadTime;
  349       FlexPWM->SUB[subMod].DTCNT1.R = DeadTime;
  350       FlexPWM->SUB[subMod].CTRL.R = prescaler | 0x0400;
  351       FlexPWM->SUB[subMod].TCTRL.B.OUT_TRIG_EN = outTrig;
  352       FlexPWM->SUB[subMod].OCTRL.R = OCTRL;
  353       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | mask;/* LDOK */
  354       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | (mask << 8);/* RUN  */
  355     }

  357     /******************************************************************************
  358      *   Function:  pwm_update_564xL_comp
  359        @brief    FlexPWM Update function for complimentary mode .
  360        @details  FlexPWM Update function for complimentary mode
  361   
  362        @param[in]     subMod   -
  363        @param[in]     Freq      -
  364        @param[in]     Duty      -
  365        @param[in]     DeadTime      -
  366        @param[in]     DeadTimeLag      -
  367   
  368        @return  none
  369      */
  370     void pwm_update_564xL_comp(uint8_t pwmMod, uint8_t SubMod, uint32_t Freq,
  371       uint16_t resolution, uint16_t Duty, uint16_t DeadTimeLead, uint16_t
  372       DeadTimeLag)
  373     {
  374       volatile mcPWM_tag* FlexPWM = FlexPWM_module[pwmMod];
  375       uint16_t val1, val2, val3;
  376       uint16_t init, mask, prescaler;
  377       uint32_t period = 0U;
  378       uint32_t DutyMax = 100 * resolution;
  379       mask = (uint16_t)(0x1U << SubMod);
  380       prescaler = pwm_prescaler (Freq, &period);
  381       val1 = (uint16_t)(period / 2);
  382       init = 0 - val1;
  383       if (Duty >= DutyMax) {
  384         val2 = init;
  385         val3 = val1+1;
  386       } else {
  387         val3 = (uint16_t)((Duty * val1)/DutyMax);
  388         val2 = 0 - val3;
  389       }
  390   
  391       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | (mask << 4);/* CLDOK */
  392       FlexPWM->SUB[SubMod].INIT.R = init;
  393       FlexPWM->SUB[SubMod].VAL[1].R = val1;
  394       FlexPWM->SUB[SubMod].VAL[2].R = val2;
  395       FlexPWM->SUB[SubMod].VAL[3].R = val3;
  396       FlexPWM->SUB[SubMod].DTCNT0.R = DeadTimeLead;
  397       FlexPWM->SUB[SubMod].DTCNT1.R = DeadTimeLag;
  398       FlexPWM->SUB[SubMod].CTRL.R = prescaler | 0x0400;
  399       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | mask;/* LDOK */
  400     }

  474       mask = (uint16_t)(0x1U << subMod);
  475       prescaler = pwm_prescaler (Freq, &period);
  476       val1 = (uint16_t)(period / 2);
  477       init = 0 - val1;
  478       val0 = 0;
  479       if (DutyA >= DutyMax) {
  480         val2 = init;
  481         val3 = val1+1;                   /* +1 */
  482       } else if (centered) {             /* center aligned PWMs */
  483         val3 = (uint16_t)((DutyA * val1)/DutyMax);
  484         val2 = 0 - val3;
  485       } else {                           /* phase shifted anf edge aligned */
  486         if (ShiftA == 0) {
  487           val2 = init;
  488         } else {
  489           while (ShiftA > DutyMax) {
  490             ShiftA = ShiftA - DutyMax;
  491           }
  492   
  493           if (ShiftA >= DutyMaxHalf) {
  494             val2 = (uint16_t)((val1 * ShiftA) / DutyMaxHalf) - val1;
  495           } else {
  496             val2 = 0 - (val1 - (uint16_t)((val1 * ShiftA) / DutyMaxHalf));
  497           }
  498         }
  499   
  500         shift_lag = ShiftA + DutyA;
  501         if (shift_lag > DutyMax) {
  502           shift_lag = shift_lag - DutyMax;
  503         }
  504   
  505         if (shift_lag >= DutyMaxHalf) {
  506           val3 = (uint16_t)((val1 * shift_lag) / DutyMaxHalf) - val1;
  507         } else {
  508           val3 = 0 - (val1 - (uint16_t)((val1 * shift_lag) / DutyMaxHalf));
  509         }
  510       }
  511   
  512       if (DutyB >= DutyMax) {
  513         val4 = init;
  514         val5 = val1+1;                   /* +1 */
  515       } else if (centered) {             /* center aligned PWMs */
  516         val5 = (uint16_t)((DutyB * val1)/DutyMax);
  517         val4 = 0 - val5;
  518       } else {                           /* phase shifted anf edge aligned */
  519         if (ShiftB == 0) {
  520           val4 = init;
  521         } else {
  522           while (ShiftB > DutyMax) {
  523             ShiftB = ShiftB - DutyMax;
  524           }
  525   
  526           if (ShiftB >= DutyMaxHalf) {
  527             val4 = (uint16_t)((val1 * ShiftB) / DutyMaxHalf) - val1;
  528           } else {
  529             val4 = 0 - (val1 - (uint16_t)((val1 * ShiftB) / DutyMaxHalf));
  530           }
  531         }
  532   
  533         shift_lag = ShiftB + DutyB;
  534         if (shift_lag > DutyMax) {
  535           shift_lag = shift_lag - DutyMax;
  536         }
  537   
  538         if (shift_lag >= DutyMaxHalf) {
  539           val5 = (uint16_t)((val1 * shift_lag) / DutyMaxHalf) - val1;
  540         } else {
  541           val5 = 0 - (val1 - (uint16_t)((val1 * shift_lag) / DutyMaxHalf));
  542         }
  543       }
  544   
  545       /* Independent PWMs */
  546       /* Initial value for PWM23, PWM45 : 1 */
  547       if (masterSync) {
  548         FlexPWM->SUB[subMod].CTRL2.R = 0xBA00;/* Master sync from submodule 0 */
  549       } else {
  550         FlexPWM->SUB[subMod].CTRL2.R = 0xB800;/* Local sync */
  551       }
  552   
  553       FlexPWM->SUB[subMod].INIT.R = init;
  554       FlexPWM->SUB[subMod].VAL[0].R = val0;
  555       FlexPWM->SUB[subMod].VAL[1].R = val1;
  556       FlexPWM->SUB[subMod].VAL[2].R = val2;
  557       FlexPWM->SUB[subMod].VAL[3].R = val3;
  558       FlexPWM->SUB[subMod].VAL[4].R = val4;
  559       FlexPWM->SUB[subMod].VAL[5].R = val5;
  560       if (doubleSw) {
  561         FlexPWM->SUB[subMod].CTRL.R = prescaler | 0x0401;/* double switching enabled  */
  562       } else {
  563         FlexPWM->SUB[subMod].CTRL.R = prescaler | 0x0400;/* double switching disabled */
  564       }
  565   
  566       FlexPWM->SUB[subMod].TCTRL.B.OUT_TRIG_EN = outTrig;
  567       FlexPWM->SUB[subMod].OCTRL.R = OCTRL;
  568       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | mask;/* LDOK */
  569       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | (uint16_t)(mask << 8);/* RUN  */
  570     }

  572     static void pwm_update_564xL_simple(uint8_t pwmMod, uint8_t subMod,
  573       FlexPWMparams_type const* params)
  574     {
  575       volatile mcPWM_tag* FlexPWM = FlexPWM_module[pwmMod];
  576       uint8_t i;
  577       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | (uint16_t)(params->mask << 4);/* CLDOK */
  578       FlexPWM->SUB[subMod].INIT.R = params->init;
  579       for (i = 0; i < 6; i++) {
  580         FlexPWM->SUB[subMod].VAL[i].R = params->val[i];
  581       }
  582   
  583       FlexPWM->SUB[subMod].CTRL.R = params->prescaler | 0x0400 | (FlexPWM->
  584         SUB[subMod].CTRL.R & 0x0001);    /* double switching */
  585       FlexPWM->MCTRL.R = FlexPWM->MCTRL.R | params->mask;/* LDOK */
  586     }
