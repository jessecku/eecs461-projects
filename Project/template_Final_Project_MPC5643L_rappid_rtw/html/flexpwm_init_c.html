   94       /* Deadtime source SELA_3 : Generated PWM signal is used by the deadtime logic.    */

  100       /* Deadtime source SELA_0 : Generated PWM signal is used by the deadtime logic.    */
  101       /* Deadtime source SELB_0 : Generated PWM signal is used by the deadtime logic.    */
  102       FlexPWM_module[pwmMod]->FSTS.B.FFLAG = 0xF;
  103   
  104       /* Clear Fault FFLAGx     */
  105       FlexPWM_module[pwmMod]->FCTRL.R = 0x0000;
  106   
  107       /* Fault Level : logic 0    */
  108       /* Automatic Fault Clearing : Disabled    */
  109       /*Fault Safe Mode         : Disabled    */
  110       /*Fault Interrupt         : Disabled    */
