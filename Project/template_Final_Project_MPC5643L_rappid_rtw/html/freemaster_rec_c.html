  252   
  253   /*!
  254    *
  255    * @brief    Handling SETUPREC and SETUPREC_EX commands
  256    *
  257    * @param    pMessageIO - original command (in) and response buffer (out)
  258    *
  259    * @return   As all command handlers, the return value should be the buffer
  260    *           pointer where the response output finished (except checksum)
  261    *
  262    ******************************************************************************/
  263   FMSTR_BPTR FMSTR_SetUpRec(FMSTR_BPTR pMessageIO)
  264   {
  265     FMSTR_BPTR pResponse = pMessageIO;
  266     FMSTR_SIZE8 nRecVarsetSize;
  267     FMSTR_SIZE blen;
  268     FMSTR_U8 i, sz;
  269     FMSTR_U8 nResponseCode;
  270   
  271     /* de-initialize first   */
  272     FMSTR_AbortRec();
  273   
  274   #if FMSTR_REC_OWNBUFF
  275   
  276     /* user wants to use his own buffer, check if it is valid */
  277     if (!pcm_nRecBuffAddr || !pcm_wRecBuffSize) {
  278       return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_INVBUFF);
  279     }
  280   
  281   #elif (FMSTR_USE_FASTREC) == 0
  282   
  283     /* size in native sizeof units (=bytes on most platforms) */
  284     FMSTR_ARR2ADDR(pcm_nRecBuffAddr, pcm_pOwnRecBuffer);
  285   
  286   #endif
  287   
  288     /* seek the setup data */
  289   #if (FMSTR_USE_FASTREC) == 0
  290   
  291     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  292     pMessageIO = FMSTR_ValueFromBuffer8(&pcm_nRecTriggerMode, pMessageIO);
  293   
  294   #else                                  /* (FMSTR_USE_FASTREC) == 0 */
  295   
  296     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 3U);
  297   
  298   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
  299   
  300     pMessageIO = FMSTR_ValueFromBuffer16(&pcm_wRecTotalSmps, pMessageIO);
  301   
  302   #if (FMSTR_REC_STATIC_POSTTRIG) == 0
  303   
  304     pMessageIO = FMSTR_ValueFromBuffer16(&pcm_wRecPostTrigger, pMessageIO);
  305   
  306   #else                                  /* (FMSTR_REC_STATIC_POSTTRIG) == 0 */
  307   
  308     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  309   
  310   #endif                                 /* (FMSTR_REC_STATIC_POSTTRIG) == 0 */
  311   
  312   #if (FMSTR_REC_STATIC_DIVISOR) == 0
  313   
  314     pMessageIO = FMSTR_ValueFromBuffer16(&pcm_wRecTimeDiv, pMessageIO);
  315   
  316   #else                                  /* (FMSTR_REC_STATIC_DIVISOR) == 0 */
  317   
  318     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  319   
  320   #endif                                 /* (FMSTR_REC_STATIC_DIVISOR) == 0 */
  321   
  322   #if (FMSTR_USE_FASTREC) == 0
  323   
  324     /* address & size of trigger variable */
  325     pMessageIO = FMSTR_AddressFromBuffer(&pcm_nTrgVarAddr, pMessageIO);
  326     pMessageIO = FMSTR_ValueFromBuffer8(&pcm_nTrgVarSize, pMessageIO);
  327   
  328     /* trigger compare mode  */
  329     pMessageIO = FMSTR_ValueFromBuffer8(&pcm_bTrgVarSigned, pMessageIO);
  330   
  331     /* threshold value  */
  332     pMessageIO = FMSTR_ValueFromBuffer32(&pcm_uTrgThreshold.u32, pMessageIO);
  333   
  334   #else                                  /* (FMSTR_USE_FASTREC) == 0 */
  335   
  336     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 8U);
  337   
  338   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
  339   
  340     /* recorder variable count */
  341     pMessageIO = FMSTR_ValueFromBuffer8(&pcm_nRecVarCount, pMessageIO);
  342   
  343     /* rec variable information must fit into our buffers */
  344     if (!pcm_nRecVarCount || pcm_nRecVarCount > (FMSTR_U8)FMSTR_MAX_REC_VARS) {
  345   
  346   #if FMSTR_REC_COMMON_ERR_CODES
  347   
  348       goto FMSTR_SetUpRec_exit_error;
  349   
  350   #else
  351   
  352       nResponseCode = FMSTR_STC_INVBUFF;
  353       goto FMSTR_SetUpRec_exit;
  354   
  355   #endif
  356   
  357     }
  358   
  359     /* calculate sum of sizes of all variables */
  360     nRecVarsetSize = 0U;
  361   
  362     /* get all addresses and sizes */
  363     for (i=0U; i<pcm_nRecVarCount; i++) {
  364       /* variable size */
  365       pMessageIO = FMSTR_ValueFromBuffer8(&sz, pMessageIO);
  366       pcm_pRecVarSize[i] = sz;
  367       nRecVarsetSize += sz;
  368   
  369       /* variable address */
  370       pMessageIO = FMSTR_AddressFromBuffer(&pcm_pRecVarAddr[i], pMessageIO);
  371   
  372       /* valid numeric variable sizes only */
  373       if (sz == 0U || sz > 8U) {
  374   
  375   #if FMSTR_REC_COMMON_ERR_CODES
  376   
  377         goto FMSTR_SetUpRec_exit_error;
  378   
  379   #else
  380   
  381         nResponseCode = FMSTR_STC_INVSIZE;
  382         goto FMSTR_SetUpRec_exit;
  383   
  384   #endif
  385   
  386       }
  387   
  388   #if FMSTR_CFG_BUS_WIDTH > 1U
  389   
  390       /* even sizes only */
  391       if (sz & 0x1) {
  392   
  393   #if FMSTR_REC_COMMON_ERR_CODES
  394   
  395         goto FMSTR_SetUpRec_exit_error;
  396   
  397   #else
  398   
  399         nResponseCode = FMSTR_STC_INVSIZE;
  400         goto FMSTR_SetUpRec_exit;
  401   
  402   #endif
  403   
  404       }
  405   
  406   #endif                                 /* FMSTR_CFG_BUS_WIDTH > 1U */
  407   
  408   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  409   
  410       if (!FMSTR_CheckTsaSpace(pcm_pRecVarAddr[i], (FMSTR_SIZE8)sz, 0U)) {
  411   
  412   #if FMSTR_REC_COMMON_ERR_CODES
  413   
  414         goto FMSTR_SetUpRec_exit_error;
  415   
  416   #else
  417   
  418         nResponseCode = FMSTR_STC_EACCESS;
  419         goto FMSTR_SetUpRec_exit;
  420   
  421   #endif
  422   
  423       }
  424   
  425   #endif                                 /* FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY */
  426   
  427     }
  428   
  429     /* fast recorder handles trigger by itself */
  430   #if (FMSTR_USE_FASTREC) == 0
  431   
  432     /* any trigger? */
  433     pcm_pCompareFunc = NULL;
  434     if (pcm_nRecTriggerMode) {
  435       /* access to trigger variable? */
  436   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  437   
  438       if (!FMSTR_CheckTsaSpace(pcm_nTrgVarAddr, (FMSTR_SIZE8)pcm_nTrgVarSize, 0U))
  439       {
  440   
  441   #if FMSTR_REC_COMMON_ERR_CODES
  442   
  443         goto FMSTR_SetUpRec_exit_error;
  444   
  445   #else
  446   
  447         nResponseCode = FMSTR_STC_EACCESS;
  448         goto FMSTR_SetUpRec_exit;
  449   
  450   #endif
  451   
  452       }
  453   
  454   #endif                                 /* FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY */
  455   
  456       /* get compare function */
  457   #if FMSTR_REC_FLOAT_TRIG
  458   
  459       if (pcm_bTrgVarSigned&FMSTR_REC_FLOAT_TRIG_MASK) {
  460         pcm_pCompareFunc = FMSTR_Comparefloat;
  461       } else
  462   #else
  463   
  464         if (pcm_bTrgVarSigned&FMSTR_REC_FLOAT_TRIG_MASK) {
  465   
  466   #if FMSTR_REC_COMMON_ERR_CODES
  467   
  468           goto FMSTR_SetUpRec_exit_error;
  469   
  470   #else
  471   
  472           nResponseCode = FMSTR_STC_FLOATDISABLED;
  473           goto FMSTR_SetUpRec_exit;
  474   
  475   #endif
  476   
  477         }
  478   
  479   #endif
  480   
  481       {
  482         switch (pcm_nTrgVarSize)
  483         {
  484   
  485   #if FMSTR_CFG_BUS_WIDTH == 1U
  486   
  487          case 1:
  488           pcm_pCompareFunc = pcm_bTrgVarSigned ? FMSTR_Compare8S : FMSTR_Compare8U;
  489           break;
  490   
  491   #endif
  492   
  493          case 2:
  494           pcm_pCompareFunc = pcm_bTrgVarSigned ? FMSTR_Compare16S :
  495             FMSTR_Compare16U;
  496           break;
  497   
  498          case 4:
  499           pcm_pCompareFunc = pcm_bTrgVarSigned ? FMSTR_Compare32S :
  500             FMSTR_Compare32U;
  501           break;
  502   
  503           /* invalid trigger variable size  */
  504          default:
  505   
  506   #if FMSTR_REC_COMMON_ERR_CODES
  507   
  508           goto FMSTR_SetUpRec_exit_error;
  509   
  510   #else
  511   
  512           nResponseCode = FMSTR_STC_INVSIZE;
  513           goto FMSTR_SetUpRec_exit;
  514   
  515   #endif
  516   
  517         }
  518       }
  519     }
  520   
  521   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
  522   
  523     /* total recorder buffer length in native sizeof units (=bytes on most platforms) */
  524     blen = (FMSTR_SIZE) (pcm_wRecTotalSmps * nRecVarsetSize / FMSTR_CFG_BUS_WIDTH);
  525   
  526     /* recorder memory available? */
  527     if (blen > FMSTR_GetRecBuffSize()) {
  528   
  529   #if FMSTR_REC_COMMON_ERR_CODES
  530   
  531       goto FMSTR_SetUpRec_exit_error;
  532   
  533   #else
  534   
  535       nResponseCode = FMSTR_STC_INVSIZE;
  536       goto FMSTR_SetUpRec_exit;
  537   
  538   #endif
  539   
  540     }
  541   
  542   #if (FMSTR_USE_FASTREC) == 0
  543   
  544     /* remember the effective end of circular buffer */
  545     pcm_dwRecEndBuffPtr = pcm_nRecBuffAddr + blen;
  546   
  547   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
  548   
  549   #if FMSTR_USE_FASTREC
  550   
  551     if (!FMSTR_SetUpFastRec()) {
  552   
  553   #if FMSTR_REC_COMMON_ERR_CODES
  554   
  555       goto FMSTR_SetUpRec_exit_error;
  556   
  557   #else                                  /* FMSTR_REC_COMMON_ERR_CODES */
  558   
  559       nResponseCode = FMSTR_STC_FASTRECERR;
  560       goto FMSTR_SetUpRec_exit;
  561   
  562   #endif                                 /* FMSTR_REC_COMMON_ERR_CODES */
  563   
  564     }
  565   
  566   #endif                                 /* FMSTR_USE_FASTREC */
  567   
  568     /* everything is okay    */
  569     pcm_wRecFlags.flg.bIsConfigured = 1U;
  570     nResponseCode = FMSTR_STS_OK;
  571   
  572   #if FMSTR_REC_COMMON_ERR_CODES
  573   
  574     goto FMSTR_SetUpRec_exit;
  575    FMSTR_SetUpRec_exit_error:
  576     nResponseCode = FMSTR_STC_INVSIZE;
  577   
  578   #endif
  579   
  580    FMSTR_SetUpRec_exit:
  581     return FMSTR_ConstToBuffer8(pResponse, nResponseCode);
  582   }

  611   /**************************************************************************/
  612   
  613   /*!
  614    *
  615    * @brief    Handling STARTREC command
  616    *
  617    * @param    pMessageIO - original command (in) and response buffer (out)
  618    *
  619    * @return   As all command handlers, the return value should be the length
  620    *           of the response filled into the buffer (including status byte)
  621    *
  622    * This function starts recording (initializes internal recording variables
  623    * and flags)
  624    *
  625    ******************************************************************************/
  626   FMSTR_BPTR FMSTR_StartRec(FMSTR_BPTR pMessageIO)
  627   {
  628     FMSTR_U8 nResponseCode;
  629   
  630     /* must be configured */
  631     if (!pcm_wRecFlags.flg.bIsConfigured) {
  632   
  633   #if FMSTR_REC_COMMON_ERR_CODES
  634   
  635       goto FMSTR_StartRec_exit_error;
  636   
  637   #else
  638   
  639       nResponseCode = FMSTR_STC_NOTINIT;
  640       goto FMSTR_StartRec_exit;
  641   
  642   #endif
  643   
  644     }
  645   
  646     /* already running ? */
  647     if (pcm_wRecFlags.flg.bIsRunning) {
  648   
  649   #if FMSTR_REC_COMMON_ERR_CODES
  650   
  651       goto FMSTR_StartRec_exit_error;
  652   
  653   #else
  654   
  655       nResponseCode = FMSTR_STS_RECRUN;
  656       goto FMSTR_StartRec_exit;
  657   
  658   #endif
  659   
  660     }
  661   
  662   #if (FMSTR_USE_FASTREC) == 0
  663   
  664     /* initialize write pointer */
  665     pcm_dwRecWritePtr = pcm_nRecBuffAddr;
  666   
  667     /* current (first) sample index */
  668     pcm_wRecBuffStartIx = 0U;
  669   
  670   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
  671   
  672     /* initialize time divisor */
  673   #if (FMSTR_REC_STATIC_DIVISOR) != 1
  674   
  675     pcm_wRecTimeDivCtr = 0U;
  676   
  677   #endif
  678   
  679     /* initiate virgin cycle */
  680     pcm_wRecFlags.flg.bIsStopping = 0U;  /* no trigger active */
  681     pcm_wRecFlags.flg.bTrgCrossActive = 0U;/* waiting for threshold crossing */
  682     pcm_wRecFlags.flg.bInvirginCycle = 1U;/* initial cycle */
  683   
  684     /* run now */
  685   
  686     /* start fast recorder */
  687   #if FMSTR_USE_FASTREC
  688   
  689     FMSTR_StartFastRec();
  690   
  691   #endif                                 /* (MSTR_USE_FASTREC */
  692   
  693     /* run now */
  694     pcm_wRecFlags.flg.bIsRunning = 1U;   /* is running now! */
  695     nResponseCode = FMSTR_STS_OK;
  696   
  697   #if FMSTR_REC_COMMON_ERR_CODES
  698   
  699     goto FMSTR_StartRec_exit;
  700    FMSTR_StartRec_exit_error:
  701     nResponseCode = FMSTR_STC_NOTINIT;
  702   
  703   #endif
  704   
  705    FMSTR_StartRec_exit:
  706     return FMSTR_ConstToBuffer8(pMessageIO, nResponseCode);
  707   }

  711   /*!
  712    *
  713    * @brief    Handling STOPREC command
  714    *
  715    * @param    pMessageIO - original command (in) and response buffer (out)
  716    *
  717    * @return   As all command handlers, the return value should be the length
  718    *           of the response filled into the buffer (including status byte)
  719    *
  720    * This function stops recording (same as manual trigger)
  721    *
  722    ******************************************************************************/
  723   FMSTR_BPTR FMSTR_StopRec(FMSTR_BPTR pMessageIO)
  724   {
  725     FMSTR_U8 nResponseCode;
  726   
  727     /* must be configured */
  728     if (!pcm_wRecFlags.flg.bIsConfigured) {
  729       nResponseCode = FMSTR_STC_NOTINIT;
  730       goto FMSTR_StopRec_exit;
  731     }
  732   
  733     /* already stopped ? */
  734     if (!pcm_wRecFlags.flg.bIsRunning) {
  735       nResponseCode = FMSTR_STS_RECDONE;
  736       goto FMSTR_StopRec_exit;
  737     }
  738   
  739     /* simulate trigger */
  740     FMSTR_TriggerRec();
  741     nResponseCode = FMSTR_STS_OK;
  742    FMSTR_StopRec_exit:
  743     return FMSTR_ConstToBuffer8(pMessageIO, nResponseCode);
  744   }

  917   /**************************************************************************/
  918   /*!
  919    *
  920    * @brief    API: Recorder worker routine - can be called from application's timer ISR
  921    *
  922    *
  923    * This returns quickly if recorder is not running, otherwise it calls quite lengthy
  924    * recorder routine which does all the recorder work (sampling, triggering)
  925    *
  926    ******************************************************************************/
  927   #if defined(FMSTR_PLATFORM_56F8xxx) || defined(FMSTR_PLATFORM_56F8xx)
  928   
  929   #pragma interrupt called
  930   
  931   #endif
  932   
  933   void FMSTR_Recorder(void)
  934   {
  935     /* recorder not active */
  936     if (!pcm_wRecFlags.flg.bIsRunning) {
  937       return ;
  938     }
  939   
  940     /* do the hard work      */
  941     FMSTR_Recorder2();
  942   }
  943   
  944   /**************************************************************************/
  945   
  946   /*!
  947    *
  948    * @brief    Recorder function called when recorder is active
  949    *
  950    ******************************************************************************/
  951   #if defined(FMSTR_PLATFORM_56F8xxx) || defined(FMSTR_PLATFORM_56F8xx)
  952   
  953   #pragma interrupt called
  954   
  955   #endif
  956   
  957   static void FMSTR_Recorder2(void)
  958   {
  959     FMSTR_SIZE8 sz;
  960     FMSTR_BOOL cmp;
  961     FMSTR_U8 i;
  962   
  963   #if (FMSTR_REC_STATIC_DIVISOR) != 1
  964   
  965     /* skip this call ? */
  966     if (pcm_wRecTimeDivCtr) {
  967       /* maybe next time... */
  968       pcm_wRecTimeDivCtr--;
  969       return;
  970     }
  971   
  972     /* re-initialize divider */
  973   #if (FMSTR_REC_STATIC_DIVISOR) == 0
  974   
  975     pcm_wRecTimeDivCtr = pcm_wRecTimeDiv;
  976   
  977   #else
  978   
  979     pcm_wRecTimeDivCtr = FMSTR_REC_STATIC_DIVISOR;
  980   
  981   #endif                                 /* (FMSTR_REC_STATIC_DIVISOR) == 0 */
  982   #endif                                 /* (FMSTR_REC_STATIC_DIVISOR) != 1 */
  983   
  984     /* take snapshot of variable values */
  985     for (i=0U; i<pcm_nRecVarCount; i++) {
  986       sz = pcm_pRecVarSize[i];
  987       FMSTR_CopyMemory(pcm_dwRecWritePtr, pcm_pRecVarAddr[i], sz);
  988       sz /= FMSTR_CFG_BUS_WIDTH;
  989       pcm_dwRecWritePtr += sz;
  990     }
  991   
  992     /* another sample taken (startIx "points" after sample just taken) */
  993     /* i.e. it points to the oldest sample */
  994     pcm_wRecBuffStartIx++;
  995   
  996     /* wrap around (circular buffer) ? */
  997     if (pcm_dwRecWritePtr >= pcm_dwRecEndBuffPtr) {
  998       pcm_dwRecWritePtr = pcm_nRecBuffAddr;
  999       pcm_wRecFlags.flg.bInvirginCycle = 0U;
 1000       pcm_wRecBuffStartIx = 0U;
 1001     }
 1002   
 1003     /* no trigger testing in virgin cycle */
 1004     if (pcm_wRecFlags.flg.bInvirginCycle) {
 1005       return;
 1006     }
 1007   
 1008     /* test trigger condition if still running */
 1009     if (!pcm_wRecFlags.flg.bIsStopping && pcm_pCompareFunc != NULL) {
 1010       /* compare trigger threshold */
 1011       cmp = pcm_pCompareFunc();
 1012   
 1013       /* negated logic (falling-edge) ? */
 1014       if (pcm_nRecTriggerMode == 2U) {
 1015         cmp = (FMSTR_BOOL) !cmp;
 1016       }
 1017   
 1018       /* above threshold ? */
 1019       if (cmp) {
 1020         /* were we at least once below threshold ? */
 1021         if (pcm_wRecFlags.flg.bTrgCrossActive) {
 1022           /* EDGE TRIGGER ! */
 1023           FMSTR_TriggerRec();
 1024         }
 1025       } else {
 1026         /* we got bellow threshold, now wait for being above threshold */
 1027         pcm_wRecFlags.flg.bTrgCrossActive = 1U;
 1028       }
 1029     }
 1030   
 1031     /* in stopping mode ? (note that this bit might have been set just above!) */
 1032     if (pcm_wRecFlags.flg.bIsStopping) {
 1033       /* count down post-trigger samples expired ? */
 1034       if (!pcm_wStoprecCountDown) {
 1035         /* STOP RECORDER */
 1036         pcm_wRecFlags.flg.bIsRunning = 0U;
 1037         return;
 1038       }
 1039   
 1040       /* perhaps next time */
 1041       pcm_wStoprecCountDown--;
 1042     }
 1043   }
 1044   
 1045   #endif                                 /* (FMSTR_USE_FASTREC) == 0 */
 1046   
 1047   #else                                  /* FMSTR_USE_RECORDER && (!FMSTR_DISABLE) */
