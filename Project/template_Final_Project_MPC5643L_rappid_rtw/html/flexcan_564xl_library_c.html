   37     /******************************************************************************
   38      *   Function:  flexcan_564xl_init
   39        @brief    Initialize FlexCAN module .
   40        @details   Initialize FlexCAN module
   41   
   42        @param[in]     flex_can     FlexCAN module
   43        @param[in]     MCR     	MCR
   44        @param[in]     CR     	CR
   45   
   46        @return status
   47      */
   48     uint8_t flexcan_564xl_init(uint8_t flex_can, uint32_t MCR, uint32_t CR)
   49     {
   50       uint8_t x = 0U;
   51       uint8_t i = 0U;
   52       uint8_t status = 0U;
   53       switch (flex_can) {
   54        case FLEXCAN_0:
   55         {
   56           CAN_0.MCR.B.MDIS = 0;
   57           CAN_0.MCR.B.SOFT_RST = 1;
   58           while (CAN_0.MCR.B.SOFT_RST ==1) {
   59           }
   60   
   61           CAN_0.MCR.B.MDIS = 1;
   62           CAN_0.CR.B.CLK_SRC = 0;
   63   
   64           /*Enable the CAN module */
   65           CAN_0.MCR.B.MDIS = 0;
   66   
   67           /* Set FRZ bit */
   68           CAN_0.MCR.B.FRZ = 1;
   69   
   70           /* Set HALT bit */
   71           CAN_0.MCR.B.HALT = 1;
   72           CAN_0.MCR.R = MCR;
   73           CAN_0.CR.R = CR;
   74   
   75           /* Set the FlexCAN Maximum Buffer Number */
   76           CAN_0.MCR.B.MAXMB = 32;
   77   
   78           /* clear memory from message buffer 0 to 15 */
   79           for (x=0; x < 32; x++) {
   80             CAN_0.BUF[x].CS.R = 0;
   81             CAN_0.BUF[x].ID.R = 0;
   82             for (i=0; i < 2; i++) {
   83               CAN_0.BUF[x].DATA.W[i] = 0;
   84             }
   85           }
   86   
   87           for (x=0 ; x<32 ; x++) {
   88             CAN_0.RXIMR[x].R = 0xFFFFFFFF;
   89           }
   90   
   91           CAN_0.IMRL.R = 0x00000000;
   92           CAN_0.MCR.B.HALT = 0;
   93           CAN_0.MCR.B.FRZ = 0;
   94   
   95           /* await synchronization (delay) */
   96           for (x=1; x < 255; x++) {
   97           }
   98   
   99           if (CAN_0.MCR.B.NOT_RDY == 1) {
  100             status = 1;
  101           } else {
  102             status = 0;
  103           }
  104           break;
  105         }
  106   
  107        case FLEXCAN_1:
  108         {
  109           CAN_1.MCR.B.MDIS = 0;
  110           CAN_1.MCR.B.SOFT_RST = 1;
  111           while (CAN_1.MCR.B.SOFT_RST ==1) {
  112           }
  113   
  114           CAN_1.MCR.B.MDIS = 1;
  115           CAN_1.CR.B.CLK_SRC = 0;
  116   
  117           /*Enable the CAN module */
  118           CAN_1.MCR.B.MDIS = 0;
  119   
  120           /* Set FRZ bit */
  121           CAN_1.MCR.B.FRZ = 1;
  122   
  123           /* Set HALT bit */
  124           CAN_1.MCR.B.HALT = 1;
  125           CAN_1.MCR.R = MCR;
  126           CAN_1.CR.R = CR;
  127   
  128           /* Set the FlexCAN Maximum Buffer Number */
  129           CAN_1.MCR.B.MAXMB = 32;
  130   
  131           /* clear memory from message buffer 0 to 15 */
  132           for (x=0; x < 32; x++) {
  133             CAN_1.BUF[x].CS.R = 0;
  134             CAN_1.BUF[x].ID.R = 0;
  135             for (i=0; i < 2; i++) {
  136               CAN_1.BUF[x].DATA.W[i] = 0;
  137             }
  138           }
  139   
  140           for (x=0 ; x<32 ; x++) {
  141             CAN_1.RXIMR[x].R = 0xFFFFFFFF;
  142           }
  143   
  144           CAN_1.IMRL.R = 0x00000000;
  145           CAN_1.MCR.B.HALT = 0;
  146           CAN_1.MCR.B.FRZ = 0;
  147   
  148           /* await synchronization (delay) */
  149           for (x=1; x < 255; x++) {
  150           }
  151   
  152           if (CAN_1.MCR.B.NOT_RDY == 1) {
  153             status = 1;
  154           } else {
  155             status = 0;
  156           }
  157           break;
  158         }
  159   
  160        default:
  161         break;
  162       }
  163   
  164       return status;
  165     }

  427     {

  431         CAN_0.BUF[BufferNo].MSG_ID.R = ptr_FLEXCAN_MsgBufferStructure->id;
  432       } else if (flex_can == FLEXCAN_1) {
  433         /* clear message buffer memory */
  434         CAN_1.BUF[BufferNo].CS.R = ptr_FLEXCAN_MsgBufferStructure->code;
  435         CAN_1.BUF[BufferNo].MSG_ID.R = ptr_FLEXCAN_MsgBufferStructure->id;

  439     /******************************************************************************
  440      *   Function:flexcan_564xl_buffer_transmit
  441        @brief    transmit  FlexCAN module buffer.
  442        @details   transmit FlexCAN module buffer
  443   
  444        @param[in]     ptr_FLEXCAN_MsgBufferStructure     pointer to structure
  445        @param[in]     BufferNo     buffer
  446        @param[in]     flex_can     FlexCAN module
  447   
  448        @return none
  449      */
  450     void flexcan_564xl_buffer_transmit(const FLEXCAN_MsgBufferStructure
  451       *ptr_FLEXCAN_MsgBufferStructure,uint8_t BufferNo,uint8_t flex_can)
  452     {
  453       uint8_t i;
  454       if (flex_can==FLEXCAN_0) {
  455         if (ptr_FLEXCAN_MsgBufferStructure->ext) {
  456           CAN_0.BUF[BufferNo].CS.B.IDE = 1;/* extended ID */
  457           CAN_0.BUF[BufferNo].ID.R = ptr_FLEXCAN_MsgBufferStructure->id;
  458         } else {
  459           CAN_0.BUF[BufferNo].CS.B.IDE = 0;/* standart ID */
  460           CAN_0.BUF[BufferNo].ID.B.STD_ID = (uint16_t)
  461             ptr_FLEXCAN_MsgBufferStructure->id;
  462         }
  463   
  464         CAN_0.BUF[BufferNo].CS.B.RTR = 0;
  465         CAN_0.BUF[BufferNo].CS.B.LENGTH = ptr_FLEXCAN_MsgBufferStructure->length;
  466         for (i=0; i < ptr_FLEXCAN_MsgBufferStructure->length; i++) {
  467           CAN_0.BUF[BufferNo].DATA.B[i] = 0;
  468           CAN_0.BUF[BufferNo].DATA.B[i] = ptr_FLEXCAN_MsgBufferStructure->data[i];
  469         }
  470   
  471         CAN_0.BUF[BufferNo].CS.B.SRR = 1;/* Tx frame */
  472         CAN_0.BUF[BufferNo].CS.B.CODE |= ptr_FLEXCAN_MsgBufferStructure->code;
  473       } else if (flex_can==FLEXCAN_1) {
  474         if (ptr_FLEXCAN_MsgBufferStructure->ext) {
  475           CAN_1.BUF[BufferNo].CS.B.IDE = 1;/* extended ID */
  476           CAN_1.BUF[BufferNo].ID.R = ptr_FLEXCAN_MsgBufferStructure->id;
  477         } else {
  478           CAN_1.BUF[BufferNo].CS.B.IDE = 0;/* standart ID */
  479           CAN_1.BUF[BufferNo].ID.B.STD_ID = (uint16_t)
  480             ptr_FLEXCAN_MsgBufferStructure->id;
  481         }
  482   
  483         CAN_1.BUF[BufferNo].CS.B.RTR = 0;
  484         CAN_1.BUF[BufferNo].CS.B.LENGTH = ptr_FLEXCAN_MsgBufferStructure->length;
  485         for (i=0; i < ptr_FLEXCAN_MsgBufferStructure->length; i++) {
  486           CAN_1.BUF[BufferNo].DATA.B[i] = 0;
  487           CAN_1.BUF[BufferNo].DATA.B[i] = ptr_FLEXCAN_MsgBufferStructure->data[i];
  488         }
  489   
  490         CAN_1.BUF[BufferNo].CS.B.SRR = 1;/* Tx frame */
  491         CAN_1.BUF[BufferNo].CS.B.CODE |= ptr_FLEXCAN_MsgBufferStructure->code;
  492       }
  493     }

  497        @brief      FlexCAN module buffer full.

  499   
  500        @param[in]     flex_can     FlexCAN module
  501        @param[in]     mb     mb

  503        @return none

  505     uint8_t flexcan_564xl_rxbuffull(uint8_t flex_can, uint8_t mb)
  506     {
  507       uint8_t ret = 0U;
  508       switch (flex_can){
  509        case FLEXCAN_0:
  510         {
  511           if (CAN_0.IFLAG1.R & (0x1U << mb))
  512             ret= 1;
  513           break;
  514         }
  515   
  516        case FLEXCAN_1:
  517         {
  518           if (CAN_1.IFLAG1.R & (0x1U << mb))
  519             ret= 1;
  520           break;
  521         }
  522   
  523        default:
  524         break;
  525       }
  526   
  527       return ret;
  528     }

  576     /******************************************************************************
  577      *	Function: flexcan_564xl_get_id
  578        @brief      returns message ID.
  579        @details   returns message ID
  580   
  581        @param[in]     flex_can     FlexCAN module
  582        @param[in]     BufferNo     BufferNo
  583   
  584        @return  id
  585      */
  586     uint32_t flexcan_564xl_get_id(uint8_t flex_can, uint8_t BufferNo)
  587     {
  588       uint32_t id = 0;
  589       if (flex_can == FLEXCAN_0) {
  590         if ((CAN_0.BUF[BufferNo].CS.B.IDE & 0x01) ==0) {
  591           /* Standard ID */
  592           id = CAN_0.BUF[BufferNo].MSG_ID.B.STD_ID;
  593         } else if ((CAN_0.BUF[BufferNo].CS.B.IDE & 0x01) !=0) {
  594           /* Extended ID */
  595           id = CAN_0.BUF[BufferNo].MSG_ID.R;
  596         }
  597       } else if (flex_can == FLEXCAN_1) {
  598         if ((CAN_1.BUF[BufferNo].CS.B.IDE & 0x01) ==0) {
  599           /* Standard ID */
  600           id = CAN_1.BUF[BufferNo].MSG_ID.B.STD_ID;
  601         } else if ((CAN_1.BUF[BufferNo].CS.B.IDE & 0x01) !=0) {

  741   
  742        @return  none
  743      */
  744     void flexcan_564xl_set_filter_id(uint8_t flex_can, uint8_t mb, uint32_t id,

  746     {
  747       switch (flex_can) {
  748        case FLEXCAN_0:
  749         {
  750           if (ext) {
  751             CAN_0.BUF[mb].CS.B.IDE = 1;  /* MB for extended ID */
  752             CAN_0.BUF[mb].ID.R = id;     /* MB ID */
  753           } else {
  754             CAN_0.BUF[mb].CS.B.IDE = 0;  /* MB for standard ID */
  755             CAN_0.BUF[mb].ID.B.STD_ID = (uint16_t)id;/* MB ID */
  756           }

  758           CAN_0.BUF[mb].CS.B.CODE = CAN_RX_EMPTY;/* Set rx MB empty */
  759           break;
  760         }
  761   
  762        case FLEXCAN_1:
  763         {
  764           if (ext) {
  765             CAN_1.BUF[mb].CS.B.IDE = 1;  /* MB for extended ID */
  766             CAN_1.BUF[mb].ID.R = id;     /* MB ID */
  767           } else {
  768             CAN_1.BUF[mb].CS.B.IDE = 0;  /* MB for standard ID */
  769             CAN_1.BUF[mb].ID.B.STD_ID = (uint16_t)id;/* MB ID */
  770           }

  772           CAN_1.BUF[mb].CS.B.CODE = CAN_RX_EMPTY;/* Set rx MB empty */
  773           break;
  774   
  775          default:
  776           break;
  777         }
  778       }
