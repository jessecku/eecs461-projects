   43   /*!
   44    *
   45    * @brief    Handling GETTSAINFO and GETTSAINFO_EX command
   46    *
   47    * @param    pMessageIO - original command (in) and response buffer (out)
   48    *
   49    * @return   As all command handlers, the return value should be the buffer
   50    *           pointer where the response output finished (except checksum)
   51    *
   52    ******************************************************************************/
   53   FMSTR_BPTR FMSTR_GetTsaInfo(FMSTR_BPTR pMessageIO)
   54   {
   55     volatile FMSTR_BPTR pResponse = pMessageIO;
   56     const FMSTR_TSA_ENTRY* pTbl;
   57     FMSTR_TSA_TINDEX nTblIndex;
   58     FMSTR_TSA_TSIZE nTblSize = 0U;
   59     FMSTR_ADDR tmpAddr;
   60     FMSTR_U16 tmp16;
   61   
   62     /* get index of table the PC is requesting   */
   63     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
   64     pMessageIO = FMSTR_ValueFromBuffer16(&nTblIndex, pMessageIO);
   65   
   66     /* TSA flags */
   67     tmp16 = FMSTR_TSA_VERSION | FMSTR_TSA_FLAGS;
   68   
   69     /* sizeof TSA table entry items */
   70     /*lint -e{506,774} constant value boolean */
   71     if ((sizeof(FMSTR_TSA_ENTRY)/4U) == 4U) {
   72       tmp16 |= FMSTR_TSA_INFO_32BIT;     /* 32bit TSA entries */
   73     }
   74   
   75     /* success, flags */
   76     pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);
   77     pResponse = FMSTR_ValueToBuffer16(pResponse, tmp16);
   78   
   79     /* get the table (or NULL if no table on given index) */
   80     pTbl = FMSTR_TsaGetTable(nTblIndex, &nTblSize);
   81     FMSTR_PTR2ADDR(tmpAddr, pTbl);
   82   
   83     /* table size in bytes */
   84     nTblSize *= FMSTR_CFG_BUS_WIDTH;
   85     pResponse = FMSTR_ValueToBuffer16(pResponse, nTblSize);
   86   
   87     /* table address */
   88     return FMSTR_AddressToBuffer(pResponse, tmpAddr);
   89   }

  103   FMSTR_BPTR FMSTR_GetStringLen(FMSTR_BPTR pMessageIO)
  104   {
  105     FMSTR_BPTR pResponse = pMessageIO;
  106     FMSTR_ADDR nAddr;
  107     FMSTR_U16 nLen = 0U;
  108     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 1U);
  109     pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);
  110     nLen = FMSTR_StrLen(nAddr);
  111   
  112     /* return strign size in bytes (even on 16bit DSP) */
  113     nLen *= FMSTR_CFG_BUS_WIDTH ;
  114     pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);
  115     return FMSTR_ValueToBuffer16(pResponse, nLen);
  116   }

  120   /*!
  121    *
  122    * @brief    Helper (inline) function for TSA memory region check
  123    *
  124    * @param    nAddrUser - address of region to be checked
  125    * @param    nSizeUser - size of region to be checked
  126    * @param    nAddrSafe - address of known "safe" region
  127    * @param    wSizeSafe - size of safe region
  128    *
  129    * @return   This function returns non-zero if given user space is safe
  130    *           (i.e. it lies in given safe space)
  131    *
  132    ******************************************************************************/
  133   #if defined(FMSTR_PLATFORM_56F8xxx) || defined(FMSTR_PLATFORM_56F8xx)
  134   
  135   /* make inline */
  136   inline FMSTR_BOOL FMSTR_CheckMemSpace(FMSTR_ADDR nAddrUser, FMSTR_SIZE8
  137     nSizeUser,
  138     FMSTR_ADDR nAddrSafe, FMSTR_SIZE wSizeSafe);
  139   inline
  140   #else
  141   
  142   /* declare function prototype */
  143   static FMSTR_BOOL FMSTR_CheckMemSpace(FMSTR_ADDR nAddrUser, FMSTR_SIZE8
  144     nSizeUser,
  145     FMSTR_ADDR nAddrSafe, FMSTR_SIZE wSizeSafe);
  146   static
  147   #endif
  148   
  149   FMSTR_BOOL FMSTR_CheckMemSpace(FMSTR_ADDR nAddrUser, FMSTR_SIZE8 nSizeUser,
  150     FMSTR_ADDR nAddrSafe, FMSTR_SIZE wSizeSafe)
  151   {
  152     FMSTR_BOOL bRet = FMSTR_FALSE;
  153   
  154   #ifdef __HCS12X__
  155   
  156     /* convert from logical to global if needed */
  157     nAddrUser = FMSTR_FixHcs12xAddr(nAddrUser);
  158     nAddrSafe = FMSTR_FixHcs12xAddr(nAddrSafe);
  159   
  160   #endif
  161   
  162     if (nAddrUser >= nAddrSafe) {
  163       bRet = (FMSTR_BOOL)
  164         (((nAddrUser + nSizeUser) <= (nAddrSafe + wSizeSafe)) ? FMSTR_TRUE :
  165          FMSTR_FALSE);
  166     }
  167   
  168     return bRet;
  169   }

  173   /*!
  174    *
  175    * @brief    Check wether given memory region is "safe" (covered by TSA)
  176    *
  177    * @param    dwAddr - address of the memory to be checked
  178    * @param    nSize  - size of the memory to be checked
  179    * @param    bWriteAccess - write access is required
  180    *
  181    * @return   This function returns non-zero if user space is safe
  182    *
  183    ******************************************************************************/
  184   FMSTR_BOOL FMSTR_CheckTsaSpace(FMSTR_ADDR dwAddr, FMSTR_SIZE8 nSize, FMSTR_BOOL
  185     bWriteAccess)
  186   {
  187     const FMSTR_TSA_ENTRY* pte;
  188     FMSTR_TSA_TINDEX nTableIndex;
  189     FMSTR_TSA_TSIZE i, cnt;
  190     FMSTR_SIZE nInfo;
  191   
  192   #if FMSTR_CFG_BUS_WIDTH >= 2U
  193   
  194     /* TSA tables use sizeof() operator which returns size in "bus-widths" (e.g. 56F8xx) */
  195     nSize = (nSize + 1) / FMSTR_CFG_BUS_WIDTH;
  196   
  197   #endif
  198   
  199     /* to be as fast as possible during normal operation,
  200        check variable entries in all tables first */
  201     for (nTableIndex=0U; (pte=FMSTR_TsaGetTable(nTableIndex, &cnt)) != NULL;
  202          nTableIndex++) {
  203       /* number of items in a table */
  204       cnt /= (FMSTR_TSA_TSIZE) sizeof(FMSTR_TSA_ENTRY);
  205   
  206       /* all table entries */
  207       for (i=0U; i<cnt; i++) {
  208         /*lint -e{506,774} condition always true/false */
  209         if (sizeof(pte->addr.p) < sizeof(pte->addr.n))
  210           nInfo = (FMSTR_SIZE) pte->info.n;
  211         else
  212           nInfo = (FMSTR_SIZE) pte->info.p;
  213   
  214         /* variable entry only (also check read-write flag) */
  215         if ((nInfo & FMSTR_TSA_INFO_VAR_FLAG) && (!bWriteAccess || (nInfo &
  216               FMSTR_TSA_INFO_RWV_FLAG))) {
  217           /* need to take the larger of the two in union (will be optimized by compiler anyway) */
  218           /*lint -e{506,774} condition always true/false */
  219           if (sizeof(pte->addr.p) < sizeof(pte->addr.n)) {
  220             if (FMSTR_CheckMemSpace(dwAddr, nSize, (FMSTR_ADDR) pte->addr.n,
  221                                     (FMSTR_SIZE) (nInfo >> 2))) {
  222               return FMSTR_TRUE;         /* access granted! */
  223             }
  224           } else {
  225             /*lint -e{923} casting pointer to long (on some architectures) */
  226             if (FMSTR_CheckMemSpace(dwAddr, nSize, (FMSTR_ADDR) pte->addr.p,
  227                                     (FMSTR_SIZE) (nInfo >> 2))) {
  228               return FMSTR_TRUE;         /* access granted! */
  229             }
  230           }
  231         }
  232   
  233         pte++;
  234       }
  235     }
  236   
  237     /* no more writable memory chunks available */
  238     if (bWriteAccess) {
  239       return FMSTR_FALSE;
  240     }
  241   
  242     /* allow reading of recorder buffer */
  243   #if FMSTR_USE_RECORDER
  244   
  245     if (FMSTR_IsInRecBuffer(dwAddr, nSize)) {
  246       return FMSTR_TRUE;
  247     }
  248   
  249   #endif
  250   
  251     /* allow reading of any C-constant string referenced in TSA tables */
  252     for (nTableIndex=0U; (pte=FMSTR_TsaGetTable(nTableIndex, &cnt)) != NULL;
  253          nTableIndex++) {
  254       FMSTR_ADDR tmpAddr;
  255   
  256       /* allow reading of the TSA table itself */
  257       FMSTR_PTR2ADDR(tmpAddr, pte);
  258       if (FMSTR_CheckMemSpace(dwAddr, nSize, tmpAddr, cnt)) {
  259         return FMSTR_TRUE;
  260       }
  261   
  262       /* number of items in a table */
  263       cnt /= (FMSTR_TSA_TSIZE) sizeof(FMSTR_TSA_ENTRY);
  264   
  265       /* all table entries */
  266       for (i=0U; i<cnt; i++) {
  267         /* system strings are always accessible at C-pointers */
  268         FMSTR_PTR2ADDR(tmpAddr, pte->name.p);
  269         if (pte->name.p) {
  270           if (FMSTR_CheckMemSpace(dwAddr, nSize, tmpAddr, FMSTR_StrLen(tmpAddr)))
  271           {
  272             return FMSTR_TRUE;
  273           }
  274         }
  275   
  276         FMSTR_PTR2ADDR(tmpAddr, pte->type.p);
  277         if (pte->type.p) {
  278           if (FMSTR_CheckMemSpace(dwAddr, nSize, tmpAddr, FMSTR_StrLen(tmpAddr)))
  279           {
  280             return FMSTR_TRUE;
  281           }
  282         }
  283   
  284         pte++;
  285       }
  286     }
  287   
  288     /* no valid TSA entry found => not-safe to access the memory */
  289     return FMSTR_FALSE;
  290   }
