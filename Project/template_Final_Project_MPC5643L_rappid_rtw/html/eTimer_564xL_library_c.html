  213     /******************************************************************************

  217   
  218        @param[in]     subMod
  219        @param[in]     channel  channel
  220        @param[in]     CNTMODE  CNTMODE
  221        ......
  222        @return none
  223      */
  224     void etimer_init_564xL(
  225       uint16_t subMod, uint16_t channel, uint16_t CNTMODE, uint16_t OUTMODE,
  226       uint16_t ONCE,
  227       uint16_t LENGTH, uint16_t DIR, uint16_t OEN, uint16_t COINIT, uint16_t STPEN,
  228       uint16_t ROC, uint16_t PRISRC, uint16_t SECSRC, uint16_t PIPS, uint16_t SIPS,
  229       uint16_t OPS, uint16_t MSTR, uint16_t VAL, uint16_t COFRC, uint16_t CLC1,
  230       uint16_t CLC2, uint16_t CMPMODE, uint16_t CPT1MODE, uint16_t CPT2MODE,
  231       uint16_t ONESHOT,
  232       uint16_t CFWM, uint16_t CNTRinit, uint16_t COMP1init, uint16_t COMP2init,
  233       uint16_t FILT,
  234       uint16_t INTDMA
  235       )
  236     {
  237       volatile mcTIMER_tag* eTimer;
  238       volatile mcTIMER_CHANNEL_tag* eTimerChannel;
  239       if (subMod == 2) {
  240         eTimer = &mcTIMER2;
  241       } else if (subMod == 1) {
  242         eTimer = &mcTIMER1;
  243       } else {
  244         eTimer = &mcTIMER0;
  245       }
  246   
  247       eTimerChannel = &(eTimer->CHANNEL[channel]);
  248       eTimerChannel->CTRL1.B.CNTMODE = 0;
  249       eTimerChannel->COMP1.R = COMP1init;
  250       eTimerChannel->CMPLD1.R = COMP1init;
  251       eTimerChannel->COMP2.R = COMP2init;
  252       eTimerChannel->CMPLD2.R = COMP2init;
  253       eTimerChannel->CNTR.R = CNTRinit;
  254       eTimerChannel->LOAD.R = CNTRinit;
  255       eTimerChannel->CTRL1.B.PRISRC = PRISRC;
  256       eTimerChannel->CTRL1.B.ONCE = ONCE;
  257       eTimerChannel->CTRL1.B.LENGTH = LENGTH;
  258       eTimerChannel->CTRL1.B.DIR = DIR;
  259       eTimerChannel->CTRL1.B.SECSRC = SECSRC;
  260       eTimerChannel->CTRL2.B.OEN = OEN;
  261       eTimerChannel->CTRL2.B.RDNT = 0;
  262       eTimerChannel->CTRL2.B.VAL = VAL;
  263       eTimerChannel->CTRL2.B.COFRC = COFRC;
  264       eTimerChannel->CTRL2.B.COINIT = COINIT;
  265       eTimerChannel->CTRL2.B.SIPS = SIPS;
  266       eTimerChannel->CTRL2.B.PIPS = PIPS;
  267       eTimerChannel->CTRL2.B.OPS = OPS;
  268       eTimerChannel->CTRL2.B.MSTR = MSTR;
  269       eTimerChannel->CTRL2.B.OUTMODE = OUTMODE;
  270       eTimerChannel->CTRL3.B.STPEN = STPEN;
  271       eTimerChannel->CTRL3.B.ROC = ROC;
