  378   
  379   #if FMSTR_USE_WRITEVARMASK && FMSTR_USE_NOEX_CMDS
  380   
  381       /* write byte with mask */
  382      case FMSTR_CMD_WRITEVAR8MASK:
  383       pResponseEnd = FMSTR_WriteVarMask(pMessageIO, 1U);
  384       break;
  385   
  386       /* write word with mask */
  387      case FMSTR_CMD_WRITEVAR16MASK:
  388       pResponseEnd = FMSTR_WriteVarMask(pMessageIO, 2U);
  389       break;
  390   
  391   #endif                                 /* FMSTR_USE_WRITEVARMASK && FMSTR_USE_NOEX_CMDS */
  392   
  393   #if FMSTR_USE_TSA
  394   
  395       /* get TSA table (force EX instead of non-EX) */
  396   #if FMSTR_USE_EX_CMDS
  397   
  398      case FMSTR_CMD_GETTSAINFO_EX:

  479     } else {
  480       /* nothing sent out */
  481       return FMSTR_FALSE;
  482     }

  487   /*!
  488    *
  489    * @brief    Handling GETINFO or GETINFO_BRIEF
  490    *
  491    * @param    pMessageIO - original command (in) and response buffer (out)
  492    *
  493    * @return   As all command handlers, the return value should be the buffer
  494    *           pointer where the response output finished (except checksum)
  495    *
  496    ******************************************************************************/
  497   FMSTR_BPTR FMSTR_GetBoardInfo(FMSTR_BPTR pMessageIO)
  498   {
  499     FMSTR_BPTR pResponse = pMessageIO;
  500     FMSTR_U16 wTmp;
  501     FMSTR_U8 *pStr;
  502     pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);
  503     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_PROT_VER));/* protVer */
  504     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_CFG_FLAGS));/* cfgFlags */
  505     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_CFG_BUS_WIDTH));/* dataBusWdt */
  506     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)
  507       (FMSTR_GLOB_VERSION_MAJOR));       /* globVerMajor */
  508     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)
  509       (FMSTR_GLOB_VERSION_MINOR));       /* globVerMinor */
  510     pResponse = FMSTR_ConstToBuffer8(pResponse, (FMSTR_U8)(FMSTR_COMM_BUFFER_SIZE));/* cmdBuffSize  */
  511   
  512     /* that is all for brief info */
  513   #if FMSTR_USE_BRIEFINFO
  514   
  515     FMSTR_UNUSED(pStr);
  516     FMSTR_UNUSED(wTmp);
  517   
  518   #else                                  /* FMSTR_USE_BRIEFINFO */
  519   #if FMSTR_USE_RECORDER
  520   
  521     /* recorder buffer size is always measured in bytes */
  522     wTmp = FMSTR_GetRecBuffSize();
  523     wTmp *= FMSTR_CFG_BUS_WIDTH;
  524   
  525     /* send size and timebase    */
  526     pResponse = FMSTR_ValueToBuffer16(pResponse, wTmp);
  527     pResponse = FMSTR_ConstToBuffer16(pResponse, (FMSTR_U16) FMSTR_REC_TIMEBASE);
  528   
  529   #else                                  /* FMSTR_USE_RECORDER */
  530   
  531     FMSTR_UNUSED(wTmp);
  532   
  533     /* recorder info zeroed */
  534     pResponse = FMSTR_ConstToBuffer16(pResponse, 0);
  535     pResponse = FMSTR_ConstToBuffer16(pResponse, 0);
  536   
  537   #endif                                 /* FMSTR_USE_RECORDER */
  538   
  539   #if FMSTR_LIGHT_VERSION
  540   
  541     FMSTR_UNUSED(pStr);
  542     pResponse = FMSTR_SkipInBuffer(pResponse, (FMSTR_U8)FMSTR_DESCR_SIZE);
  543   
  544   #else
  545   
  546     /* description string */
  547     pStr = (FMSTR_U8*) FMSTR_IDT_STRING;
  548     for (wTmp = 0U; wTmp < (FMSTR_U8)(FMSTR_DESCR_SIZE); wTmp++) {
  549       pResponse = FMSTR_ValueToBuffer8(pResponse, *pStr);
  550   
  551       /* terminating zero used to clear the remainder of the buffer */
  552       if (*pStr) {
  553         pStr ++;
  554       }
  555     }
  556   
  557   #endif                                 /* SEND_IDT_STRING */
  558   #endif                                 /* FMSTR_USE_BRIEFINFO */
  559   
  560     return pResponse;
  561   }

  563   /**************************************************************************/
  564   
  565   /*!
  566    *
  567    * @brief    Handling READMEM and READMEM_EX commands
  568    *
  569    * @param    pMessageIO - original command (in) and response buffer (out)
  570    *
  571    * @return   As all command handlers, the return value should be the buffer
  572    *           pointer where the response output finished (except checksum)
  573    *
  574    ******************************************************************************/
  575   FMSTR_BPTR FMSTR_ReadMem(FMSTR_BPTR pMessageIO)
  576   {
  577     FMSTR_BPTR pResponse = pMessageIO;
  578     FMSTR_ADDR nAddr;
  579     FMSTR_U8 nSize;
  580     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  581     pMessageIO = FMSTR_ValueFromBuffer8(&nSize, pMessageIO);
  582     pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);
  583   
  584   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  585   
  586     if (!FMSTR_CheckTsaSpace(nAddr, (FMSTR_SIZE8) nSize, FMSTR_FALSE)) {
  587       return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_EACCESS);
  588     }
  589   
  590   #endif
  591   
  592     /* check the response will safely fit into comm buffer */
  593     if (nSize > (FMSTR_U8)FMSTR_COMM_BUFFER_SIZE) {
  594       return FMSTR_ConstToBuffer8(pResponse, FMSTR_STC_RSPBUFFOVF);
  595     }
  596   
  597     /* success  */
  598     pResponse = FMSTR_ConstToBuffer8(pResponse, FMSTR_STS_OK);
  599     return FMSTR_CopyToBuffer(pResponse, nAddr, (FMSTR_SIZE8) nSize);
  600   }

  640    * @param    pMessageIO - original command (in) and response buffer (out)

  642    * @return   As all command handlers, the return value should be the buffer
  643    *           pointer where the response output finished (except checksum)

  645    ******************************************************************************/
  646   FMSTR_BPTR FMSTR_WriteMem(FMSTR_BPTR pMessageIO)
  647   {
  648     FMSTR_BPTR pResponse = pMessageIO;
  649     FMSTR_ADDR nAddr;
  650     FMSTR_U8 nSize,nResponseCode;
  651     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  652     pMessageIO = FMSTR_ValueFromBuffer8(&nSize, pMessageIO);
  653     pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);
  654   
  655   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  656   
  657     if (!FMSTR_CheckTsaSpace(nAddr, (FMSTR_SIZE8) nSize, FMSTR_TRUE)) {
  658       nResponseCode = FMSTR_STC_EACCESS;
  659       goto FMSTR_WriteMem_exit;
  660     }
  661   
  662   #endif
  663   
  664     /*lint -e{534} ignoring function return value */
  665     FMSTR_CopyFromBuffer(nAddr, pMessageIO, (FMSTR_SIZE8) nSize);
  666     nResponseCode = FMSTR_STS_OK;
  667   
  668   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  669   
  670    FMSTR_WriteMem_exit:
  671   
  672   #endif
  673   
  674     return FMSTR_ConstToBuffer8(pResponse, nResponseCode);
  675   }

  698   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  699   
  700     if (!FMSTR_CheckTsaSpace(nAddr, nSize, FMSTR_TRUE)) {
  701       nResponseCode = FMSTR_STC_EACCESS;
  702       goto FMSTR_WriteVar_exit;
  703     }
  704   
  705   #endif
  706   
  707     /*lint -e{534} ignoring function return value */
  708     FMSTR_CopyFromBuffer(nAddr, pMessageIO, nSize);
  709     nResponseCode = FMSTR_STS_OK;
  710   
  711   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  712   
  713    FMSTR_WriteVar_exit:
  714   
  715   #endif
  716   
  717     return FMSTR_ConstToBuffer8(pResponse, nResponseCode);
  718   }
  719   
  720   /**************************************************************************/
  721   
  722   /*!
  723    *
  724    * @brief    Handling WRITEMEMMASK and WRITEMEMMASK_EX commands
  725    *
  726    * @param    pMessageIO - original command (in) and response buffer (out)
  727    *
  728    * @return   As all command handlers, the return value should be the buffer
  729    *           pointer where the response output finished (except checksum)
  730    *
  731    ******************************************************************************/
  732   FMSTR_BPTR FMSTR_WriteMemMask(FMSTR_BPTR pMessageIO)
  733   {
  734     FMSTR_BPTR pResponse = pMessageIO;
  735     FMSTR_ADDR nAddr;
  736     FMSTR_U8 nSize,nResponseCode;
  737     pMessageIO = FMSTR_SkipInBuffer(pMessageIO, 2U);
  738     pMessageIO = FMSTR_ValueFromBuffer8(&nSize, pMessageIO);
  739     pMessageIO = FMSTR_AddressFromBuffer(&nAddr, pMessageIO);
  740   
  741   #if FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY
  742   
  743     if (!FMSTR_CheckTsaSpace(nAddr, (FMSTR_SIZE8)nSize, FMSTR_TRUE)) {
  744       nResponseCode = FMSTR_STC_EACCESS;
  745       goto FMSTR_WriteMemMask_exit;
  746     }
  747   
  748   #endif
  749   
  750   #if FMSTR_CFG_BUS_WIDTH > 1U
  751   
  752     /* size must be divisible by bus width (mask must not begin in half of memory word) */
  753     if (nSize % FMSTR_CFG_BUS_WIDTH) {
  754       nResponseCode = FMSTR_STC_INVSIZE;
  755       goto FMSTR_WriteMemMask_exit;
  756     }
  757   
  758   #endif
  759   
  760     /* put the data */
  761     FMSTR_CopyFromBufferWithMask(nAddr, pMessageIO, (FMSTR_SIZE8)nSize);
  762     nResponseCode = FMSTR_STS_OK;
  763   
  764   #if (FMSTR_USE_TSA && FMSTR_USE_TSA_SAFETY) || (FMSTR_CFG_BUS_WIDTH > 1U)
  765   
  766    FMSTR_WriteMemMask_exit:
  767   
  768   #endif
  769   
  770     return FMSTR_ConstToBuffer8(pResponse, nResponseCode);
  771   }
