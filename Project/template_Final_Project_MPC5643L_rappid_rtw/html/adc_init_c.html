   33     /*********************  Initialization Function(s) ************************/
   34     /******************************************************************************
   35      *   Function:  adc_init_fnc
   36        @brief    ADC Initialization Function .
   37        @details       ADC Initialization Function
   38   
   39        @return none
   40      */
   41     void adc_init_fnc(void)
   42     {
   43       /*-----------------------------------------------------------*/
   44       /*            Abort all conversion in process                */
   45       /*-----------------------------------------------------------*/
   46       ADC_0.MCR.B.ABORT = 1;
   47       ADC_1.MCR.B.ABORT = 1;
   48   
   49       /*-----------------------------------------------------------*/
   50       /*            ADC Sampling and ThresHold Configuration                    */
   51       /*-----------------------------------------------------------*/
   52       adc_sampling_setup_fnc();
   53   
   54       /*-----------------------------------------------------------*/
   55       /*            ADC General Setup                    */
   56       /*-----------------------------------------------------------*/
   57       adc_general_setup_fnc();
   58   
   59       /*-----------------------------------------------------------*/
   60       /*            ADC CTU Configuration        */
   61       /*-----------------------------------------------------------*/
   62       adc_ctu_fnc();
   63   
   64       /*-----------------------------------------------------------*/
   65       /*            ADC Interrupt Configuration                            */
   66       /*-----------------------------------------------------------*/
   67       adc_interrupt_setup_fnc();
   68   
   69       /*-----------------------------------------------------------*/
   70       /*            ADC Channel Configuration                    */
   71       /*-----------------------------------------------------------*/
   72       adc_channel_setup_fnc();
   73   
   74       /*-----------------------------------------------------------*/
   75       /*            ADC Trigger Configuration                    */
   76       /*-----------------------------------------------------------*/
   77       adc_trigger_setup_fnc();
   78   
   79       /*-----------------------------------------------------------*/
   80       /*            DMA Setup Function(s)                    */
   81       /*-----------------------------------------------------------*/
   82       adc_dma_setup_fnc();
   83   
   84       /*-----------------------------------------------------------*/
   85       /*            Exit the ABORT state                    */
   86       /*-----------------------------------------------------------*/
   87       ADC_0.MCR.B.ABORT = 0;
   88       ADC_1.MCR.B.ABORT = 0;
   89     }

  155     /*********************  Sampling Initialization Function(s) ************************/
  156     /******************************************************************************
  157      *   Function:  adc_sampling_setup_fnc
  158        @brief    ADC Sampling Initialization Function .
  159        @details       ADC Sampling Initialization Function
  160   
  161        @return none
  162      */
  163     void adc_sampling_setup_fnc()
  164     {
  165       /*-----------------------------------------------------------*/
  166       /*                ADC_0 Sampling Control            */
  167       /*-----------------------------------------------------------*/
  168   
  169       /*-----------------------------------------------------------*/
  170       /*                Conversion Timing Register 0(ADC_0_CTR0)    */
  171       /*-----------------------------------------------------------*/
  172       ADC_0.CTR[0].R = 0x00008203;
  173   
  174       /*Phase duration Latch : Enabled */
  175       /*Input Sampling Duration : 3 clock Cycles     */
  176       /*Input Comparison Duration : 1 clock Cycles     */
  177   
  178       /*-----------------------------------------------------------*/
  179       /*                Offset  Word Register (ADC_OFFWR)    */
  180       /*-----------------------------------------------------------*/
  181       //        ADC_0.OFFWR.B.OFFSETWORD = 0x00000000;
  182       /* Offset Loading : Disabled     */
  183       /* External Offset Word : 0    */
  184   
  185       /*-----------------------------------------------------------*/
  186       /*             Main Configuration Register (ADC_0_MCR)    */
  187       /*-----------------------------------------------------------*/
  188       ADC_0.MCR.R = 0x00000000;
  189   
  190       /* Offset Cancellation Phase : Disabled */
  191       /* Offset Refresh during Idle mode :Disabled     */
  192       ADC_0.MCR.B.PWDN = 0x0;
  193   
  194       /* Exit from power down state*/
  195       while (ADC_0.MCR.B.OFFCANC) {
  196         /*No Operation*/
  197       }
  198   
  199       /* Note that offset cancellation is performed only after coming
  200          out of powerdown state. As a consequence, OFFCANC bit must be
  201          set before PWDN bit resetting after the offset cancellation
  202          phase is over the 8-bit offset coefficient is stored in the
  203          Offset word register OFFREG and as a consequence, the OFFCANC bit is cleared */
  204   
  205       /*-----------------------------------------------------------*/
  206       /*                ADC_1 Sampling Control            */
  207       /*-----------------------------------------------------------*/
  208   
  209       /*-----------------------------------------------------------*/
  210       /*                Conversion Timing Register 0(ADC_1_CTR0)    */
  211       /*-----------------------------------------------------------*/
  212       ADC_1.CTR[0].R = 0x00008203;
  213   
  214       /*Phase duration Latch : Enabled */
  215       /*Input Sampling Duration : 3 clock Cycles     */
  216       /*Input Comparison Duration : 1 clock Cycles     */
  217   
  218       /*-----------------------------------------------------------*/
  219       /*                Offset  Word Register (ADC_OFFWR)    */
  220       /*-----------------------------------------------------------*/
  221       //        ADC_1.OFFWR.B.OFFSETWORD = 0x00000000;
  222       /* Offset Loading : Disabled     */
  223       /* External Offset Word : 0    */
  224   
  225       /*-----------------------------------------------------------*/
  226       /*             Main Configuration Register (ADC_1_MCR)    */
  227       /*-----------------------------------------------------------*/
  228       ADC_1.MCR.R = 0x00000000;
  229   
  230       /* Offset Cancellation Phase : Disabled */
  231       /* Offset Refresh during Idle mode :Disabled     */
  232       ADC_1.MCR.B.PWDN = 0x0;
  233   
  234       /* Exit from power down state*/
  235       while (ADC_1.MCR.B.OFFCANC) {
  236         /*No Operation*/
  237       }
  238   
  239       /* Note that offset cancellation is performed only after coming
  240          out of powerdown state. As a consequence, OFFCANC bit must be
  241          set before PWDN bit resetting after the offset cancellation
  242          phase is over the 8-bit offset coefficient is stored in the
  243          Offset word register OFFREG and as a consequence, the OFFCANC bit is cleared */
  244     }
